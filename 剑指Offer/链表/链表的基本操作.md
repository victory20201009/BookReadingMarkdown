# 2 向链表末尾添加元素

```C++
void addToTail(ListNode** pHead,int v){
    ListNode* pNode = new ListNode();
    pNode->value=v;
    pNode->next=NULL;
    if(*pHead==NULL){
        *pHead = pNode;
    }else{
        ListNode* pTmp = *pHead;
        while(pTmp->next!=NULL)
            pTmp=pTmp->next;
        pTmp->next=pNode;
    }
}
```

# 3 根据条件删除链表中第一个元素

```C++
void removeFirst(ListNode** pHead,int n){
    if(pHead==NULL||*pHead==NULL)
        return;
    ListNode* p2DeteleNode=NULL;
    if((*pHead)->value==n){
        p2DeteleNode = *pHead;
        *pHead = (*pHead)->next;
    }else{
        ListNode* pTmp = *pHead;
        while(pTmp->next!=NULL&&pTmp->next->value!=n)
            pTmp=pTmp->next;
        if(pTmp->next!=NULL){
            p2DeteleNode = pTmp->next;
            pTmp->next=p2DeteleNode->next;
        }
    }
    if(p2DeteleNode!=NULL){
        delete p2DeteleNode;
        p2DeteleNode=NULL;
    }
}
```

# 4 倒叙打印链表

- 思路1：使用栈，正序入栈，在一个一个弹出即可

  ```C++
  void printListReversingly(ListNode* head){
      std::stack<ListNode*> pStack;
      ListNode* pNode=head;
      while(pNode!=NULL){
          pStack.push(pNode);
          pNode=pNode->next;
      }
      while(!pStack.empty()){
          pNode = pStack.top();
          printf("%d\t",pNode->value);
          pStack.pop();
      }
  }
  ```

- 思路2：递归。当打印当前元素时，先打印其后面的元素

  ```C++
  void printListReversingly(ListNode* head){
      if(head!=NULL){
          /*当它后面还有元素的时候，先输出其后面的*/
          if(head->next!=NULL)
          	printListReversingly(head->next);
      	printf("%d\t",head->value);
      }
  }
  ```
